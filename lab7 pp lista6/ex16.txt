#include <stdio.h>
#include <stdlib.h>

int main()
{  
  char str[1000];
  int i, j, prod, max_prod = 0;

  fgets(str, 1000, stdin); 

  for (i = 0; str[i + 4] != '\0' && str[i + 4] != '\n'; i++)
  {
    prod = 1;
    for (j = i; j < i + 5; j++)
    {
      prod *= (str[j] - 48);
    }
    if (prod > max_prod)
    {
      max_prod = prod;
    }
  }
  printf("O maior produto Ã© %d\n", max_prod); 
  return 0;
}
